import Lottie from 'lottie-react'
import bell from './bell.json'

export function Bell() {
  return (
    <Lottie
      animationData={bell}
      loop
      autoplay
    />
  )
}
<div id="bell"></div>

<script>
  lottie.loadAnimation({
    container:
      document.getElementById('bell'),
    path: './bell.json',
    loop: true,
    autoplay: true
  })
</script>
import Lottie

let bell = LottieAnimationView(
  name: "bell"
)
bell.loopMode = .loop
bell.play()

view.addSubview(bell)
