Drop into any frontend

React
import Lottie from 'lottie-react'
import bell from './bell.json'

export function Bell() {
  return (
    <Lottie
      animationData={bell}
      loop
      autoplay
    />
  )
}
Vanilla JS
<div id="bell"></div>

<script>
  lottie.loadAnimation({
    container:
      document.getElementById('bell'),
    path: './bell.json',
    loop: true,
    autoplay: true
  })
</script>
Swift
import Lottie

let bell = LottieAnimationView(
  name: "bell"
)
bell.loopMode = .loop
bell.play()

view.addSubview(bell)
